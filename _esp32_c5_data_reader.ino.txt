/*
  ESP32-C5 多传感器数据读取 - 串口输出版本
  数据格式: JSON，每行一个JSON对象
  可供Python读取并可视化

  硬件: ESP32-C5-DevKitM-1
  连接:
  - 传感器A0 -> GPIO1 (ADC1_CH0)
  - 传感器A1 -> GPIO2 (ADC1_CH1)
  - 可根据需要修改引脚定义

  串口波特率: 115200
*/

#include <Arduino.h>

// 引脚定义 - 根据你的实际连接修改
#define ADC_PIN_1 1   // ADC1_CH0
#define ADC_PIN_2 2   // ADC1_CH1

// 内置温度传感器
#if CONFIG_IDF_TARGET_ESP32C5
#include <driver/temperature_sensor.h>
temperature_sensor_handle_t temp_handle = NULL;
#endif

// 数据采集间隔 (毫秒)
const int SAMPLE_INTERVAL = 100;

void setup() {
  Serial.begin(115200);
  delay(1000);

  // 初始化ADC
  analogReadResolution(12);  // 12位分辨率，0-4095

  // 初始化内置温度传感器 (ESP32-C5)
  #if CONFIG_IDF_TARGET_ESP32C5
  temperature_sensor_config_t temp_config = TEMPERATURE_SENSOR_CONFIG_DEFAULT(10, 50);
  if (temperature_sensor_install(&temp_config, &temp_handle) != ESP_OK) {
    Serial.println("{\"error\":\"温度传感器初始化失败\"}");
  } else {
    temperature_sensor_enable(temp_handle);
  }
  #endif

  Serial.println("ESP32-C5 Data Reader Ready");
  Serial.println("Format: JSON per line");
}

float readInternalTemperature() {
  #if CONFIG_IDF_TARGET_ESP32C5
  if (temp_handle == NULL) return NAN;
  float temp;
  if (temperature_sensor_get_celsius(temp_handle, &temp) != ESP_OK) {
    return NAN;
  }
  return temp;
  #else
  return NAN;
  #endif
}

void loop() {
  static unsigned long last_sample = 0;
  unsigned long now = millis();

  if (now - last_sample >= SAMPLE_INTERVAL) {
    last_sample = now;

    // 读取ADC值
    int adc1 = analogRead(ADC_PIN_1);
    int adc2 = analogRead(ADC_PIN_2);

    // 转换为电压 (0-3.3V)
    float voltage1 = (adc1 * 3.3f) / 4095.0f;
    float voltage2 = (adc2 * 3.3f) / 4095.0f;

    // 读取内部温度
    float internal_temp = readInternalTemperature();

    // 输出JSON格式
    Serial.print("{");
    Serial.print("\"time\":");
    Serial.print(now);
    Serial.print(",\"adc1_raw\":");
    Serial.print(adc1);
    Serial.print(",\"adc1_volt\":");
    Serial.print(voltage1, 3);
    Serial.print(",\"adc2_raw\":");
    Serial.print(adc2);
    Serial.print(",\"adc2_volt\":");
    Serial.print(voltage2, 3);
    if (!isnan(internal_temp)) {
      Serial.print(",\"internal_temp\":");
      Serial.print(internal_temp, 2);
    }
    Serial.print("}\n");
  }

  delay(10);
}
